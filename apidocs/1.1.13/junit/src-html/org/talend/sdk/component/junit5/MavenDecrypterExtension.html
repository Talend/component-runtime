016package org.talend.sdk.component.junit5;
017
018import java.lang.annotation.Annotation;
019
020import org.junit.jupiter.api.extension.ExtensionContext;
021import org.junit.jupiter.api.extension.ParameterResolutionException;
022import org.talend.sdk.component.api.DecryptedServer;
023import org.talend.sdk.component.base.BaseMavenDecrypter;
024import org.talend.sdk.component.junit.base.junit5.JUnit5InjectionSupport;
025import org.talend.sdk.component.maven.Server;
026
027public class MavenDecrypterExtension implements JUnit5InjectionSupport {
028
029    private final BaseMavenDecrypter decrypter = new BaseMavenDecrypter();
030
031    @Override
032    public Class<? extends Annotation> injectionMarker() {
033        return DecryptedServer.class;
034    }
035
036    @Override
037    public boolean supports(final Class<?> type) {
038        return Server.class == type;
039    }
040
041    @Override
042    public Object findInstance(final ExtensionContext extensionContext, final Class<?> type, final Annotation marker)
043            throws ParameterResolutionException {
044        return decrypter.createInstance(DecryptedServer.class.cast(marker));
045    }
046}
