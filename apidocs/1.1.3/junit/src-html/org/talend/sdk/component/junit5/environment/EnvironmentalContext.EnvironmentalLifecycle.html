016package org.talend.sdk.component.junit5.environment;
017
018import static java.util.Collections.singletonList;
019import static java.util.Optional.ofNullable;
020
021import java.util.List;
022
023import org.junit.jupiter.api.extension.AfterEachCallback;
024import org.junit.jupiter.api.extension.BeforeEachCallback;
025import org.junit.jupiter.api.extension.ConditionEvaluationResult;
026import org.junit.jupiter.api.extension.ExecutionCondition;
027import org.junit.jupiter.api.extension.Extension;
028import org.junit.jupiter.api.extension.ExtensionContext;
029import org.junit.jupiter.api.extension.TestTemplateInvocationContext;
030import org.talend.sdk.component.junit.environment.DecoratingEnvironmentProvider;
031import org.talend.sdk.component.junit.environment.EnvironmentProvider;
032
033import lombok.AllArgsConstructor;
034
035@AllArgsConstructor
036public class EnvironmentalContext implements TestTemplateInvocationContext {
037
038    private final EnvironmentProvider provider;
039
040    private final String displayName;
041
042    @Override
043    public String getDisplayName(final int invocationIndex) {
044        return displayName;
045    }
046
047    @Override
048    public List<Extension> getAdditionalExtensions() {
049        return singletonList(new EnvironmentalLifecycle(provider, null));
050    }
051
052    @AllArgsConstructor
053    public static class EnvironmentalLifecycle implements BeforeEachCallback, AfterEachCallback, ExecutionCondition {
054
055        private final EnvironmentProvider provider;
056
057        private AutoCloseable closeable;
058
059        @Override
060        public void beforeEach(final ExtensionContext context) {
061            closeable = provider.start(context.getRequiredTestClass(), context.getRequiredTestClass().getAnnotations());
062        }
063
064        @Override
065        public void afterEach(final ExtensionContext context) {
066            ofNullable(closeable).ifPresent(c -> {
067                try {
068                    c.close();
069                } catch (final Exception e) {
070                    throw new IllegalStateException(e);
071                }
072            });
073        }
074
075        @Override
076        public ConditionEvaluationResult evaluateExecutionCondition(final ExtensionContext context) {
077            return isActive() ? ConditionEvaluationResult.enabled("provider is active")
078                    : ConditionEvaluationResult.disabled("provider is disabled");
079        }
080
081        private boolean isActive() {
082            return DecoratingEnvironmentProvider.class.isInstance(provider)
083                    && DecoratingEnvironmentProvider.class.cast(provider).isActive();
084        }
085    }
086}
