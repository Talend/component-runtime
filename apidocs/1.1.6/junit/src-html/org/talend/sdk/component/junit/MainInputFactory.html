016package org.talend.sdk.component.junit;
017
018import static java.util.Collections.singletonMap;
019
020import java.io.Serializable;
021import java.util.Iterator;
022
023import lombok.Data;
024
025/**
026 * Holds data for the __default__ input branch only. All other branches will
027 * return null.
028 */
029@Data
030public class MainInputFactory implements ControllableInputFactory {
031
032    private final Iterator<Serializable> main;
033
034    @Override
035    public Object read(final String name) {
036        return "__default__".equals(name) ? (main.hasNext() ? main.next() : null) : null;
037    }
038
039    @Override
040    public boolean hasMoreData() {
041        return main.hasNext();
042    }
043
044    @Override
045    public InputFactoryIterable asInputRecords() {
046        return new InputFactoryIterable(this, singletonMap("__default__", main));
047    }
048}
