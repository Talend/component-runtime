149    enum Type {
150
151        RECORD(new Class<?>[] { Record.class }),
152        ARRAY(new Class<?>[] { Collection.class }),
153        STRING(new Class<?>[] { String.class }),
154        BYTES(new Class<?>[] { byte[].class, Byte[].class }),
155        INT(new Class<?>[] { Integer.class }),
156        LONG(new Class<?>[] { Long.class }),
157        FLOAT(new Class<?>[] { Float.class }),
158        DOUBLE(new Class<?>[] { Double.class }),
159        BOOLEAN(new Class<?>[] { Boolean.class }),
160        DATETIME(new Class<?>[] { Long.class, Date.class, Temporal.class });
161
162        /** All compatibles Java classes */
163        private final Class<?>[] classes;
164
165        Type(final Class<?>[] classes) {
166            this.classes = classes;
167        }
168
169        /**
170         * Check if input can be affected to an entry of this type.
171         * 
172         * @param input : object.
173         * @return true if input is null or ok.
174         */
175        public boolean isCompatible(final Object input) {
176            if (input == null) {
177                return true;
178            }
179            for (final Class<?> clazz : classes) {
180                if (clazz.isInstance(input)) {
181                    return true;
182                }
183            }
184            return false;
185        }

491    /**
492     * Sanitize name to be avro compatible.
493     * 
494     * @param name : original name.
495     * @return avro compatible name.
496     */
497    static String sanitizeConnectionName(final String name) {
498        if (name == null || name.isEmpty()) {
499            return name;
500        }
501
502        char current = name.charAt(0);
503        final CharsetEncoder ascii = Charset.forName(StandardCharsets.US_ASCII.name()).newEncoder();
504        final boolean skipFirstChar = ((!ascii.canEncode(current)) || (!Character.isLetter(current) && current != '_'))
505                && name.length() > 1 && (!Character.isDigit(name.charAt(1)));
506
507        final StringBuilder sanitizedBuilder = new StringBuilder();
508
509        if (!skipFirstChar) {
510            if (((!Character.isLetter(current)) && current != '_') || (!ascii.canEncode(current))) {
511                sanitizedBuilder.append('_');
512            } else {
513                sanitizedBuilder.append(current);
514            }
515        }
516        for (int i = 1; i < name.length(); i++) {
517            current = name.charAt(i);
518            if (!ascii.canEncode(current)) {
519                if (Character.isLowerCase(current) || Character.isUpperCase(current)) {
520                    sanitizedBuilder.append('_');
521                } else {
522                    final byte[] encoded =
523                            Base64.getEncoder().encode(name.substring(i, i + 1).getBytes(StandardCharsets.UTF_8));
524                    final String enc = new String(encoded);
525                    if (sanitizedBuilder.length() == 0 && Character.isDigit(enc.charAt(0))) {
526                        sanitizedBuilder.append('_');
527                    }
528                    for (int iter = 0; iter < enc.length(); iter++) {
529                        if (Character.isLetterOrDigit(enc.charAt(iter))) {
530                            sanitizedBuilder.append(enc.charAt(iter));
531                        } else {
532                            sanitizedBuilder.append('_');
533                        }
534                    }
535                }
536            } else if (Character.isLetterOrDigit(current)) {
537                sanitizedBuilder.append(current);
538            } else {
539                sanitizedBuilder.append('_');
540            }
541
542        }
543        return sanitizedBuilder.toString();
544    }

546    @AllArgsConstructor
547    @ToString
548    @EqualsAndHashCode
549    class EntriesOrder implements Comparator<Entry> {
550
551        @Getter
552        private final List<String> fieldsOrder;
553
554        /**
555         * Build an EntriesOrder according fields.
556         *
557         * @param fields the fields ordering
558         * @return the order EntriesOrder
559         */
560        public static EntriesOrder of(final String fields) {
561            return new EntriesOrder(fields);
562        }
563
564        public EntriesOrder(final String fields) {
565            if (fields == null) {
566                fieldsOrder = emptyList();
567            } else {
568                fieldsOrder = Arrays.stream(fields.split(",")).collect(Collectors.toList());
569            }
570        }
571
572        /**
573         * Move a field after another one.
574         *
575         * @param after the field name reference
576         * @param name the field name
577         * @return this EntriesOrder
578         */
579        public EntriesOrder moveAfter(final String after, final String name) {
580            if (getFieldsOrder().indexOf(after) == -1) {
581                throw new IllegalArgumentException(String.format("%s not in schema", after));
582            }
583            getFieldsOrder().remove(name);
584            int destination = getFieldsOrder().indexOf(after);
585            if (!(destination + 1 == getFieldsOrder().size())) {
586                destination += 1;
587            }
588            getFieldsOrder().add(destination, name);
589            return this;
590        }
591
592        /**
593         * Move a field before another one.
594         *
595         * @param before the field name reference
596         * @param name the field name
597         * @return this EntriesOrder
598         */
599        public EntriesOrder moveBefore(final String before, final String name) {
600            if (getFieldsOrder().indexOf(before) == -1) {
601                throw new IllegalArgumentException(String.format("%s not in schema", before));
602            }
603            getFieldsOrder().remove(name);
604            getFieldsOrder().add(getFieldsOrder().indexOf(before), name);
605            return this;
606        }
607
608        /**
609         * Swap two fields.
610         *
611         * @param name the field name
612         * @param with the other field
613         * @return this EntriesOrder
614         */
615        public EntriesOrder swap(final String name, final String with) {
616            Collections.swap(getFieldsOrder(), getFieldsOrder().indexOf(name), getFieldsOrder().indexOf(with));
617            return this;
618        }
619
620        public String toFields() {
621            return getFieldsOrder().stream().collect(Collectors.joining(","));
622        }
623
624        @Override
625        public int compare(final Entry e1, final Entry e2) {
626            final int index1 = getFieldsOrder().indexOf(e1.getName());
627            final int index2 = getFieldsOrder().indexOf(e2.getName());
628            if (index1 >= 0 && index2 >= 0) {
629                return index1 - index2;
630            }
631            if (index1 >= 0) {
632                return -1;
633            }
634            if (index2 >= 0) {
635                return 1;
636            }
637            return 0;
638        }
639    }

641    static Schema.Entry avoidCollision(final Schema.Entry newEntry,
642            final Supplier<Stream<Schema.Entry>> allEntriesSupplier, final BiConsumer<String, Entry> replaceFunction) {
643        final Optional<Entry> collisionedEntry = allEntriesSupplier //
644                .get() //
645                .filter((final Entry field) -> field.getName().equals(newEntry.getName())
646                        && !Objects.equals(field, newEntry)) //
647                .findFirst();
648        if (!collisionedEntry.isPresent()) {
649            // No collision, return new entry.
650            return newEntry;
651        }
652        final Entry matchedEntry = collisionedEntry.get();
653        final boolean matchedToChange = matchedEntry.getRawName() != null && !(matchedEntry.getRawName().isEmpty());
654        if (matchedToChange) {
655            // the rename has to be applied on entry already inside schema, so replace.
656            replaceFunction.accept(matchedEntry.getName(), newEntry);
657        } else if (newEntry.getRawName() == null || newEntry.getRawName().isEmpty()) {
658            // try to add exactly same raw, skip the add here.
659            return null;
660        }
661        final Entry fieldToChange = matchedToChange ? matchedEntry : newEntry;
662        int indexForAnticollision = 1;
663        final String baseName = Schema.sanitizeConnectionName(fieldToChange.rawName); // recalc primiti name.
664
665        String newName = baseName + "_" + indexForAnticollision;
666        final Set<String> existingNames = allEntriesSupplier //
667                .get() //
668                .map(Entry::getName) //
669                .collect(Collectors.toSet());
670        while (existingNames.contains(newName)) {
671            indexForAnticollision++;
672            newName = baseName + "_" + indexForAnticollision;
673        }
674        final Entry newFieldToAdd = fieldToChange.toBuilder().withName(newName).build();
675
676        return newFieldToAdd; // matchedToChange ? newFieldToAdd : newEntry;
677    }
