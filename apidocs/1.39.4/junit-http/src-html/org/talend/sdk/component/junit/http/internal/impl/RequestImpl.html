016package org.talend.sdk.component.junit.http.internal.impl;
017
018import java.util.Map;
019
020import org.talend.sdk.component.junit.http.api.Request;
021
022import lombok.AllArgsConstructor;
023
024@AllArgsConstructor
025public class RequestImpl implements Request {
026
027    private final String uri;
028
029    private final String method;
030
031    private final String payload;
032
033    private final Map<String, String> headers;
034
035    @Override
036    public String method() {
037        return method;
038    }
039
040    @Override
041    public Map<String, String> headers() {
042        return headers;
043    }
044
045    @Override
046    public String uri() {
047        return uri;
048    }
049
050    @Override
051    public String payload() {
052        return payload;
053    }
054}
