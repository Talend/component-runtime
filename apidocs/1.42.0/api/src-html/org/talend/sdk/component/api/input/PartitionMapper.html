016package org.talend.sdk.component.api.input;
017
018import static java.lang.annotation.ElementType.TYPE;
019import static java.lang.annotation.RetentionPolicy.RUNTIME;
020
021import java.lang.annotation.Retention;
022import java.lang.annotation.Target;
023
024@Target(TYPE)
025@Retention(RUNTIME)
026public @interface PartitionMapper {
027
028    /**
029     * @return the component of this mapper (family/grouping value).
030     */
031    String family() default "";
032
033    /**
034     * @return the value of the related input.
035     */
036    String name() default "";
037
038    /**
039     * If the @Producer method of the underlying {@link Emitter} can return null before the data
040     * are completely read - i.e. infinite flow case - then you should set true to prevent the
041     * execution to complete when null is encountered.
042     *
043     * Default cases matches a batch case whereas when set to true it matches a stream case.
044     *
045     * @return true if the underlying input can be used as a stream and not in batch context.
046     */
047    boolean infinite() default false;
048}
