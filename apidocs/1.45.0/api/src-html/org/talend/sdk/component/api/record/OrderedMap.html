016package org.talend.sdk.component.api.record;
017
018import java.util.Collections;
019import java.util.HashMap;
020import java.util.Iterator;
021import java.util.Map;
022import java.util.NoSuchElementException;
023import java.util.Optional;
024import java.util.function.Consumer;
025import java.util.function.Function;
026import java.util.stream.Stream;
027import java.util.stream.StreamSupport;
028
029import lombok.AllArgsConstructor;
030import lombok.Getter;
031
032/**
033 * On TCK Record, we have to control order of elements with keeping efficient access.
034 * LinkedHashMap has efficient access, but order of element is only the order of insertions.
035 * List (ArrayList, LinkedList) allow to fine control order but have inefficent access.
036 * This class aims to control element order with keeping efficient access.
037 * 
038 * @param <T> : type of element.
039 */
040public class OrderedMap<T> {
041
042    @AllArgsConstructor
043    static class Node<T> implements Iterable<Node<T>> {
044
045        @Getter
046        public T entry;
047
048        public Node<T> next;
049
050        public Node<T> prec;
051
052        public Node<T> insert(final T newEntry) {
053            final Node<T> newNode = new Node<>(newEntry, this.next, this);
054            return this.insert(newNode);
055        }
056
057        public Node<T> insert(final Node<T> newNode) {
058            if (newNode == this) {
059                return newNode;
060            }
061            if (this.next != null) {
062                this.next.prec = newNode;
063            }
064            newNode.prec = this;
065            newNode.next = this.next;
066            this.next = newNode;
067
068            return newNode;
069        }
070
071        public void remove() {
072            if (next != null) {
073                this.next.prec = this.prec;
074            }
075            if (this.prec != null) {
076                this.prec.next = this.next;
077            }
078            this.next = null;
079            this.prec = null;
080        }
081
082        @Override
083        public Iterator<Node<T>> iterator() {
084            return new EntryNodeIterator(this);
085        }
086
087        @AllArgsConstructor
088        static class EntryNodeIterator<T> implements Iterator<Node<T>> {
089
090            private Node<T> current;
091
092            @Override
093            public boolean hasNext() {
094                return current != null;
095            }
096
097            @Override
098            public Node<T> next() {
099                if (!this.hasNext()) {
100                    throw new NoSuchElementException("no further node");
101                }
102                final Node<T> next = this.current;
103                this.current = next.next;
104                return next;
105            }
106        }
107    }
108
109    private Node<T> first;
110
111    private Node<T> last;
112
113    private final Map<String, Node<T>> entryIndex;
114
115    private final Function<T, String> getName;
116
117    public OrderedMap(final Function<T, String> getName) {
118        this(getName, Collections.emptyList());
119    }
120
121    public OrderedMap(final Function<T, String> getName,
122            final Iterable<T> inputEntries) {
123        this.entryIndex = new HashMap<>();
124        this.getName = getName;
125        inputEntries.forEach(this::addEntry);
126    }
127
128    public Stream<T> getEntries() {
129        if (this.first == null) {
130            return Stream.empty();
131        }
132        final Function<Node<T>, T> tFunction = (Node<T> f) -> f.getEntry();
133        return StreamSupport.stream(this.first.spliterator(), false)
134                .map(tFunction);
135    }
136
137    public void forEachEntry(final Consumer<T> entryConsumer) {
138        if (this.first != null) {
139            this.first.forEach((Node<T> node) -> entryConsumer.accept(node.getEntry()));
140        }
141    }
142
143    public void removeEntry(final T schemaEntry) {
144        final String name = this.getName.apply(schemaEntry);
145        final Node<T> entry = this.entryIndex.remove(name);
146        if (entry == null) {
147            throw new IllegalArgumentException(
148                    "No entry '" + name + "' expected in entries");
149        }
150        this.removeFromChain(entry);
151    }
152
153    private void removeFromChain(final Node<T> node) {
154        if (this.first == node) {
155            this.first = node.next;
156        }
157        if (this.last == node) {
158            this.last = node.prec;
159        }
160        node.remove();
161    }
162
163    public T getEntry(final String name) {
164        if (this.entryIndex != null) {
165            return Optional.ofNullable(this.entryIndex.get(name)) //
166                    .map(Node::getEntry) //
167                    .orElse(null);
168        } else {
169            return null;
170        }
171    }
172
173    public void replaceEntry(final String name, final T newEntry) {
174        final Node<T> entryNode = this.entryIndex.remove(name);
175        if (entryNode != null) {
176            entryNode.entry = newEntry;
177            final String newEntryName = this.getName.apply(newEntry);
178            this.entryIndex.put(newEntryName, entryNode);
179        }
180    }
181
182    public void addEntry(final T entry) {
183        final String name = this.getName.apply(entry);
184        if (this.entryIndex != null && this.entryIndex.containsKey(name)) {
185            return;
186        }
187        if (this.first == null) {
188            this.first = new Node<>(entry, null, null);
189            this.last = this.first;
190            this.entryIndex.put(name, this.first);
191        } else {
192            final Node<T> newNode = this.last.insert(entry);
193            this.entryIndex.put(name, newNode);
194            this.last = newNode;
195        }
196    }
197
198    public void moveAfter(final String name, final T newEntry) {
199        final Node<T> entryPivot = this.entryIndex.get(name);
200        if (entryPivot == null) {
201            throw new IllegalArgumentException(String.format("%s not in schema", name));
202        }
203        String entryName = this.getName.apply(newEntry);
204        final Node<T> entryToMove = this.entryIndex.get(entryName);
205
206        this.removeFromChain(entryToMove);
207        entryPivot.insert(entryToMove);
208        if (entryPivot == this.last) {
209            this.last = entryToMove;
210        }
211    }
212
213    /**
214     * New Entry should take place before 'name' entry
215     *
216     * @param name : entry to move before (pivot).
217     * @param newEntry : entry to move.
218     */
219    public void moveBefore(final String name, final T newEntry) {
220        final Node<T> entryPivot = this.entryIndex.get(name);
221        if (entryPivot == null) {
222            throw new IllegalArgumentException(String.format("%s not in schema", name));
223        }
224        final String newEntryName = this.getName.apply(newEntry);
225        final Node<T> entryToMove = this.entryIndex.get(newEntryName);
226
227        this.removeFromChain(entryToMove);
228        if (entryPivot == this.first) {
229            entryToMove.next = entryPivot;
230            entryPivot.prec = entryToMove;
231            this.first = entryToMove;
232        } else {
233            entryPivot.prec.insert(entryToMove);
234        }
235    }
236
237    public void swap(final String first, final String second) {
238        if (first == null || second == null || first.equals(second)) {
239            return;
240        }
241        final Node<T> firstNode = this.entryIndex.get(first);
242        if (firstNode == null) {
243            throw new IllegalArgumentException(String.format("%s not in schema", first));
244        }
245        final Node<T> secondNode = this.entryIndex.get(second);
246        if (secondNode == null) {
247            throw new IllegalArgumentException(String.format("%s not in schema", secondNode));
248        }
249        final Node<T> firstPrec = firstNode.prec;
250        final Node<T> firstNext = firstNode.next;
251
252        final Node<T> secondPrec = secondNode.prec;
253        final Node<T> secondNext = secondNode.next;
254
255        if (secondPrec == firstNode) { // case first -> second direct
256            secondNode.next = firstNode;
257            secondNode.prec = firstPrec;
258            if (firstPrec != null) {
259                firstPrec.next = secondNode;
260            }
261            firstNode.prec = secondNode;
262            firstNode.next = secondNext;
263            if (secondNext != null) {
264                secondNext.prec = firstNode;
265            }
266        } else if (firstPrec == secondNode) { // case second -> first direct
267            firstNode.next = secondNode;
268            firstNode.prec = secondPrec;
269            if (secondPrec != null) {
270                secondPrec.next = firstNode;
271            }
272            secondNode.prec = firstNode;
273            secondNode.next = firstNext;
274            if (firstNext != null) {
275                firstNext.prec = secondNode;
276            }
277        } else { // general case
278                 // Put first node at second place
279            firstNode.prec = secondPrec;
280            if (secondPrec != null) {
281                secondPrec.next = firstNode;
282            }
283            firstNode.next = secondNext;
284            if (secondNext != null) {
285                secondNext.prec = firstNode;
286            }
287
288            // Put second node at first place
289            secondNode.prec = firstPrec;
290            if (firstPrec != null) {
291                firstPrec.next = secondNode;
292            }
293            secondNode.next = firstNext;
294            if (firstNext != null) {
295                firstNext.prec = secondNode;
296            }
297        }
298        if (this.first == firstNode) {
299            this.first = secondNode;
300        } else if (this.first == secondNode) {
301            this.first = firstNode;
302        }
303        if (this.last == firstNode) {
304            this.last = secondNode;
305        } else if (this.last == secondNode) {
306            this.last = firstNode;
307        }
308    }
