016package org.talend.sdk.component.api.service.healthcheck;
017
018import static java.lang.annotation.ElementType.METHOD;
019import static java.lang.annotation.RetentionPolicy.RUNTIME;
020
021import java.lang.annotation.Retention;
022import java.lang.annotation.Target;
023
024import org.talend.sdk.component.api.meta.Documentation;
025import org.talend.sdk.component.api.service.ActionType;
026
027/**
028 * Mark a method as retruning the {@link HealthCheckStatus} of a datastore. The only configuration
029 * parameter will be an {@link org.talend.sdk.component.api.configuration.Option} named "datastore"
030 * and matching the @DataStore property.
031 */
032@ActionType(value = "healthcheck", expectedReturnedType = HealthCheckStatus.class)
033@Target(METHOD)
034@Retention(RUNTIME)
035@Documentation("This class marks an action doing a connection test")
036public @interface HealthCheck {
037
038    /**
039     * @return the value of the component family this action relates to.
040     */
041    String family() default "";
042
043    /**
044     * @return the value of the action.
045     */
046    String value() default "default";
047}
