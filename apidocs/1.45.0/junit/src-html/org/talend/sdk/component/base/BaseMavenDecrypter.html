016package org.talend.sdk.component.base;
017
018import java.util.stream.Stream;
019
020import org.talend.sdk.component.api.DecryptedServer;
021import org.talend.sdk.component.maven.MavenDecrypter;
022import org.talend.sdk.component.maven.Server;
023
024public class BaseMavenDecrypter {
025
026    public Server createInstance(final DecryptedServer config) {
027        if (config.alwaysTryLookup()) {
028            try {
029                return new MavenDecrypter().find(config.value());
030            } catch (final IllegalArgumentException iae) {
031                if (ifActive(config.conditions())) {
032                    throw iae;
033                }
034            }
035        } else if (ifActive(config.conditions())) {
036            return new MavenDecrypter().find(config.value());
037        }
038        final Server server = new Server();
039        server.setUsername(config.defaultUsername());
040        server.setPassword(config.defaultPassword());
041        return server;
042    }
043
044    private boolean ifActive(final DecryptedServer.Conditions conditions) {
045        final Stream<Boolean> evaluations = Stream.of(conditions.value()).map(c -> {
046            final String value = System.getProperty(c.forSystemProperty());
047            if (value == null) {
048                return c.supportsNull();
049            } else {
050                return c.expectedValue().equals(value);
051            }
052        });
053        switch (conditions.combination()) {
054        case AND:
055            return evaluations.allMatch(c -> c);
056        case OR:
057            return evaluations.anyMatch(c -> c);
058        default:
059            throw new IllegalArgumentException("Unsupported: " + conditions.combination());
060        }
061    }
062}
