016package org.talend.sdk.component.api.record;
017
018import java.util.Collections;
019import java.util.HashMap;
020import java.util.Iterator;
021import java.util.Map;
022import java.util.NoSuchElementException;
023import java.util.Optional;
024import java.util.function.Consumer;
025import java.util.function.Function;
026import java.util.stream.Stream;
027import java.util.stream.StreamSupport;
028
029import lombok.AllArgsConstructor;
030import lombok.Getter;
031
032/**
033 * On TCK Record, we have to control order of elements with keeping efficient access.
034 * LinkedHashMap has efficient access, but order of element is only the order of insertions.
035 * List (ArrayList, LinkedList) allow to fine control order but have inefficent access.
036 * This class aims to control element order with keeping efficient access.
037 * 
038 * @param <T> : type of element.
039 */
040public class OrderedMap<T> {
041
042    @AllArgsConstructor
043    static class Node<T> implements Iterable<Node<T>> {
044
045        @Getter
046        public T value;
047
048        public Node<T> next;
049
050        public Node<T> prec;
051
052        public Node<T> insert(final T newValue) {
053            final Node<T> newNode = new Node<>(newValue, this.next, this);
054            return this.insert(newNode);
055        }
056
057        public Node<T> insert(final Node<T> newNode) {
058            if (newNode == this) {
059                return newNode;
060            }
061            if (this.next != null) {
062                this.next.prec = newNode;
063            }
064            newNode.prec = this;
065            newNode.next = this.next;
066            this.next = newNode;
067
068            return newNode;
069        }
070
071        public void remove() {
072            if (next != null) {
073                this.next.prec = this.prec;
074            }
075            if (this.prec != null) {
076                this.prec.next = this.next;
077            }
078            this.next = null;
079            this.prec = null;
080        }
081
082        /**
083         * switch current node with its next
084         * For example : 1 <=> 2(this) <=> 3 <=> 4
085         * will be transformed to : 1 <=> 3 <=> 2(this) <=> 4
086         */
087        public void swapWithNext() {
088            final Node<T> nextNode = this.next;
089            if (nextNode == null) {
090                return;
091            }
092            // save what will be changed
093            final Node<T> firstPrec = this.prec;
094            final Node<T> nextNext = nextNode.next;
095
096            // change next node
097            nextNode.next = this;
098            nextNode.prec = this.prec;
099
100            // change prec.next to new next
101            if (firstPrec != null) {
102                firstPrec.next = nextNode;
103            }
104            this.prec = nextNode;
105            this.next = nextNext;
106            if (nextNext != null) {
107                nextNext.prec = this;
108            }
109        }
110
111        @Override
112        public Iterator<Node<T>> iterator() {
113            return new NodeIterator(this);
114        }
115
116        @AllArgsConstructor
117        static class NodeIterator<T> implements Iterator<Node<T>> {
118
119            private Node<T> current;
120
121            @Override
122            public boolean hasNext() {
123                return current != null;
124            }
125
126            @Override
127            public Node<T> next() {
128                if (!this.hasNext()) {
129                    throw new NoSuchElementException("no further node");
130                }
131                final Node<T> next = this.current;
132                this.current = next.next;
133                return next;
134            }
135        }
136    }
137
138    private Node<T> first;
139
140    private Node<T> last;
141
142    private final Map<String, Node<T>> values;
143
144    private final Function<T, String> identifierGetter;
145
146    public OrderedMap(final Function<T, String> identifierGetter) {
147        this(identifierGetter, Collections.emptyList());
148    }
149
150    public OrderedMap(final Function<T, String> identifierGetter,
151            final Iterable<T> inputValues) {
152        this.values = new HashMap<>();
153        this.identifierGetter = identifierGetter;
154        inputValues.forEach(this::addValue);
155    }
156
157    public Stream<T> streams() {
158        if (this.first == null) {
159            return Stream.empty();
160        }
161        final Function<Node<T>, T> tFunction = (Node<T> f) -> f.getValue();
162        return StreamSupport.stream(this.first.spliterator(), false)
163                .map(tFunction);
164    }
165
166    public void forEachValue(final Consumer<T> valueConsumer) {
167        if (this.first != null) {
168            this.first.forEach((Node<T> node) -> valueConsumer.accept(node.getValue()));
169        }
170    }
171
172    public void removeValue(final T value) {
173        final String identifier = this.identifierGetter.apply(value);
174        final Node<T> node = this.values.remove(identifier);
175        if (node == null) {
176            throw new IllegalArgumentException(
177                    "No node '" + identifier + "' expected in values");
178        }
179        this.removeFromChain(node);
180    }
181
182    private void removeFromChain(final Node<T> node) {
183        if (this.first == node) {
184            this.first = node.next;
185        }
186        if (this.last == node) {
187            this.last = node.prec;
188        }
189        node.remove();
190    }
191
192    public T getValue(final String identifier) {
193        if (this.values != null) {
194            return Optional.ofNullable(this.values.get(identifier)) //
195                    .map(Node::getValue) //
196                    .orElse(null);
197        } else {
198            return null;
199        }
200    }
201
202    public void replace(final String identifier, final T newValue) {
203        final Node<T> node = this.values.remove(identifier);
204        if (node != null) {
205            node.value = newValue;
206            final String newIdentifier = this.identifierGetter.apply(newValue);
207            this.values.put(newIdentifier, node);
208        }
209    }
210
211    public void addValue(final T value) {
212        final String name = this.identifierGetter.apply(value);
213        if (this.values != null && this.values.containsKey(name)) {
214            return;
215        }
216        if (this.first == null) {
217            this.first = new Node<>(value, null, null);
218            this.last = this.first;
219            this.values.put(name, this.first);
220        } else {
221            final Node<T> newNode = this.last.insert(value);
222            this.values.put(name, newNode);
223            this.last = newNode;
224        }
225    }
226
227    public void moveAfter(final String pivotIdentifier, final T valueToMove) {
228        final Node<T> pivotNode = this.values.get(pivotIdentifier);
229        if (pivotNode == null) {
230            throw new IllegalArgumentException(String.format("%s not in schema", pivotIdentifier));
231        }
232        final String identifier = this.identifierGetter.apply(valueToMove);
233        final Node<T> nodeToMove = this.values.get(identifier);
234
235        this.removeFromChain(nodeToMove);
236        pivotNode.insert(nodeToMove);
237        if (pivotNode == this.last) {
238            this.last = nodeToMove;
239        }
240    }
241
242    /**
243     * New Value should take place before 'pivotIdentifier' value
244     *
245     * @param pivotIdentifier : value pivotIdentifier to move before (pivot).
246     * @param valueToMove : value to move.
247     */
248    public void moveBefore(final String pivotIdentifier, final T valueToMove) {
249        final Node<T> nodePivot = this.values.get(pivotIdentifier);
250        if (nodePivot == null) {
251            throw new IllegalArgumentException(String.format("%s not in schema", pivotIdentifier));
252        }
253        final String newValueName = this.identifierGetter.apply(valueToMove);
254        final Node<T> nodeToMove = this.values.get(newValueName);
255
256        this.removeFromChain(nodeToMove);
257        if (nodePivot == this.first) {
258            nodeToMove.next = nodePivot;
259            nodePivot.prec = nodeToMove;
260            this.first = nodeToMove;
261        } else {
262            nodePivot.prec.insert(nodeToMove);
263        }
264    }
265
266    public void swap(final String first, final String second) {
267        if (first == null || second == null || first.equals(second)) {
268            return;
269        }
270        final Node<T> firstNode = this.values.get(first);
271        if (firstNode == null) {
272            throw new IllegalArgumentException(String.format("%s not in schema", first));
273        }
274        final Node<T> secondNode = this.values.get(second);
275        if (secondNode == null) {
276            throw new IllegalArgumentException(String.format("%s not in schema", secondNode));
277        }
278
279        if (firstNode.next == secondNode) { // case first -> second direct
280            firstNode.swapWithNext();
281        } else if (secondNode.next == firstNode) { // case second -> first direct
282            secondNode.swapWithNext();
283        } else { // general case
284                 // Put first node at second place
285            final Node<T> firstPrec = firstNode.prec;
286            final Node<T> firstNext = firstNode.next;
287
288            final Node<T> secondPrec = secondNode.prec;
289            final Node<T> secondNext = secondNode.next;
290
291            firstNode.prec = secondPrec;
292            if (secondPrec != null) {
293                secondPrec.next = firstNode;
294            }
295            firstNode.next = secondNext;
296            if (secondNext != null) {
297                secondNext.prec = firstNode;
298            }
299
300            // Put second node at first place
301            secondNode.prec = firstPrec;
302            if (firstPrec != null) {
303                firstPrec.next = secondNode;
304            }
305            secondNode.next = firstNext;
306            if (firstNext != null) {
307                firstNext.prec = secondNode;
308            }
309        }
310        if (this.first == firstNode) {
311            this.first = secondNode;
312        } else if (this.first == secondNode) {
313            this.first = firstNode;
314        }
315        if (this.last == firstNode) {
316            this.last = secondNode;
317        } else if (this.last == secondNode) {
318            this.last = firstNode;
319        }
320    }
