016package org.talend.sdk.component.api.configuration.action.meta;
017
018import static java.lang.annotation.ElementType.ANNOTATION_TYPE;
019import static java.lang.annotation.RetentionPolicy.RUNTIME;
020
021import java.lang.annotation.Retention;
022import java.lang.annotation.Target;
023
024@Target(ANNOTATION_TYPE)
025@Retention(RUNTIME)
026public @interface ActionRef {
027
028    /**
029     * @return action annotation marker.
030     */
031    Class<?> value();
032
033    /**
034     * @return the name of the method in the decorated annotation containing the reference to use
035     * to identify the related action.
036     */
037    String ref() default "value";
038}
