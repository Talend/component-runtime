016package org.talend.sdk.component.api.service.discovery;
017
018import java.util.ArrayList;
019import java.util.List;
020import java.util.Map;
021import java.util.TreeMap;
022
023import lombok.Getter;
024
025@Getter
026public class DiscoverDatasetResult {
027
028    private final List<DatasetDescription> datasetDescriptionList = new ArrayList<>();
029
030    public DiscoverDatasetResult() {
031    }
032
033    public DiscoverDatasetResult(final List<DatasetDescription> datasetDescriptionList) {
034        this.datasetDescriptionList.addAll(datasetDescriptionList);
035    }
036
037    @Getter
038    public final static class DatasetDescription {
039
040        public DatasetDescription(final String name) {
041            this.name = name;
042        }
043
044        private final String name;
045
046        private Map<String, String> metadata = new TreeMap<>();
047
048        public void addMetadata(final String key, final String value) {
049            this.metadata.put(key, value);
050        }
051    }
052
053}
